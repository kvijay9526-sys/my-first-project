2. Smart Pointers (No Memory Leaks)

Stop using raw pointers! Here is how to use std::unique_ptr for automatic memory management in modern C++.

#include <iostream>
#include <memory>

class Entity {
public:
    Entity() { std::cout << "Created Entity!\n"; }
    ~Entity() { std::cout << "Destroyed Entity!\n"; }
    void print() { std::cout << "Hello from Entity\n"; }
};

int main() {
    {
        // Automatically destroyed when it goes out of scope
        std::unique_ptr<Entity> entity = std::make_unique<Entity>();
        entity->print();
    } 
    // Memory is automatically freed here. No 'delete' keyword needed!
    
    return 0;
}

3. Linked List Node Implementation

The foundational building block for advanced data structures using standard structs.

#include <iostream>

// Define the Node structure
struct Node {
    int data;
    Node* next;
    
    // Constructor to initialize
    Node(int val) {
        data = val;
        next = nullptr;
    }
};

void printList(Node* head) {
    Node* temp = head;
    while (temp != nullptr) {
        std::cout << temp->data << " -> ";
        temp = temp->next;
    }
    std::cout << "NULL\n";
}

/* Usage:
Node* head = new Node(10);
head->next = new Node(20);
printList(head);
